Size

Width: Height:

size: {{wid + "*" + hei}}

<div ng-app="" ng-controller="sizeController"> Width: <input type="number" ng-model="wid" placeholder="enter the width"> Height: <input type="number" ng-model="hei" placeholder="enter the height"> <br /><br /> <img ng-model="imsize" src="../../images/image/about_me.jpg" width={{wid}} height={{hei}}> size: {{wid + "*" + hei}}<br /> </div> <script> function sizeController($scope) { $scope.wid = "140"; $scope.hei = "140"; } </script>
